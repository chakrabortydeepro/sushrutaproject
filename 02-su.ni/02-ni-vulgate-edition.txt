      <div n="2" type="sthāna">

        <head>nidānasthānam</head>

        <div n="1" type="adhyāya">
          <head>prathamo 'dhyāyaḥ |</head>
          <l xml:id="SS.2.1.1"> athāto vātavyādhinidānaṃ vyākhyāsyāmaḥ || </l>


          <l xml:id="SS.2.1.2"> yathovāca bhagavān dhanvantariḥ || </l>


          <l xml:id="SS.2.1.3"> dhanvantariṃ dharmabhṛtāṃ variṣṭham amṛtodbhavam | caraṇāv
            upasaṃgṛhya suśrutaḥ paripṛcchati || </l>


          <l xml:id="SS.2.1.4"> vāyoḥ prakṛtibhūtasya vyāpannasya ca kopanaiḥ | sthānaṃ
            karma ca rogāṃś ca vada me vadatāṃ vara || </l>

          <l xml:id="SS.2.1.5"> tasya tad vacanaṃ prābravīd bhiṣajaṃ varaḥ | svayaṃbhūr
            eṣa bhagavān vāyur ity abhiśabditaḥ || </l>

          <l xml:id="SS.2.1.6"> svātantryān nityabhāvāc ca sarvagatvāt tathaiva ca |
            sarveṣām eva sarvātmā sarvalokanamaskṛtaḥ || </l>

          <l xml:id="SS.2.1.7"> sthityutpattivināśeṣu bhūtānām eṣa kāraṇam | avyakto
            vyaktakarmā ca rūkṣaḥ śīto laghuḥ kharaḥ || </l>

          <l xml:id="SS.2.1.8"> tiryaggo dviguṇaś caiva rajo<pc> </pc>bahula eva ca |
            acintyavīryo doṣāṇāṃ netā rogasamūharāṭ || </l>

          <l xml:id="SS.2.1.9"> āśukārī muhuścārī pakvādhānagudālayaḥ | dehe vicaratas
            tasya lakṣaṇāni nibodha me || </l>

          <l xml:id="SS.2.1.10a"> doṣadhātvagni<pc> </pc>samatāṃ saṃprāptiṃ viṣayeṣu ca |
            kriyāṇām ānulomyaṃ ca karoty akupito 'nilaḥ || | </l>

          <l xml:id="SS.2.1.10"> indriyārthopasaṃprāptiṃ doṣadhātvagny<pc> </pc>avaikṛtam
            | kriyāṇām ānulomyaṃ ca kuryād vāyur adūṣitaḥ || </l>

          <l xml:id="SS.2.1.11"> yathā 'gniḥ pañcadhā bhinno nāmasthānakriyāmayaiḥ |
            bhinno 'nilas tasthā hy eko nāmasthānakriyāmayaiḥ || </l>

          <l xml:id="SS.2.1.12"> prāṇodānau samānaś ca vyānaś cāpāna eva ca | sthānasthā
            mārutāḥ ` yāpayanti śarīriṇam || </l>

          <l xml:id="SS.2.1.13"> yo vāyur vaktrasaṃcārī sa prāṇo nāma dehadhṛk | so 'nnaṃ
            praveśayaty antaḥ prāṇāṃś cāpy avalambate || </l>

          <l xml:id="SS.2.1.14"> prāyaśaḥ kurute duṣṭo hikkāśvāsādikān gadān || udāno nāma
            yas tūrdhvam upaiti pavanottamaḥ || </l>

          <l xml:id="SS.2.1.15"> tena bhāṣitagītādiviśeṣo 'bhipravartate |
            ūrdhvajatrugatān rogān karoti ca viśeṣataḥ || </l>


          <l xml:id="SS.2.1.16"> āmapakvāśayacaraḥ samāno vahnisaṅgataḥ | so 'nnaṃ pacati
            tajjāṃś ca viśeṣān vivinakti hi || </l>

          <l xml:id="SS.2.1.17"> gulmāgni<pc> </pc>sādātīsāraprabhṛtīn kurute gadān |
            kṛtsnadehacaro vyāno rasasaṃvahanodyataḥ || </l>

          <l xml:id="SS.2.1.18"> svedāsṛksrāvaṇaś cāpi pañcadhā ceṣṭayaty api | kruddhaś
            ca kurute rogān prāyaśaḥ sarvadehagān || </l>

          <l xml:id="SS.2.1.19"> pakvādhānālayo 'pānaḥ kāle karṣati cāpy adhaḥ | samīraṇaḥ
            śakṛnmūtraṃ śukragarbhārtavāni ca || </l>

          <l xml:id="SS.2.1.20"> kruddhaś ca kurute rogān ghorān bastigudāśrayān |
            śukradoṣapramehās tu vyānāpānaprakopajāḥ || </l>

          <l xml:id="SS.2.1.21ab"> yugapat kupitāś cāpi dehaṃ bhindyur asaṃśayam |</l>
          
          <l xml:id="SS.2.1.21cd"> ata ūrdhvaṃ pravakṣyāmi nānāsthānāntarāśritaḥ || </l>

          <l xml:id="SS.2.1.22"> bahuśaḥ kupito vāyur vikārān kurute hi yān | vāyur
            āmāśaye kruddhaś chardyādīn kurute gadān || </l>

          <l xml:id="SS.2.1.23"> mohaṃ mūrcchāṃ pipāsāṃ ca hṛdgrahaṃ pārśvavedanām |
            pakvāśayastho 'ntrakūjaṃ śūlaṃ nābhau karoti ca || </l>

          <l xml:id="SS.2.1.24"> kṛcchramūtrapurīṣatvam ānāhaṃ trikavedanām | śrotrādiṣv
            indriyavadhaṃ kuryāt kruddhaḥ samīraṇaḥ || </l>

          <l xml:id="SS.2.1.25"> vaivarṇyaṃ sphuraṇaṃ raukṣyaṃ suptiṃ cumucumāyanam |
            tvakstho nistodanaṃ kuryāt tvagbhedaṃ paripoṭanam || </l>

          <l xml:id="SS.2.1.26"> vraṇāṃś ca raktago granthīn saśūlān māṃsasaṃśritaḥ |
            tathā medaḥśritaḥ kuryād grandhīnmandarujo 'vraṇān || </l>

          <l xml:id="SS.2.1.27"> kuryāt sirāgataḥ śūlaṃ sirākuñcanapūraṇam | snāyuprāptaḥ
            stambhakampau śūlamākṣepaṇaṃ tathā || </l>

          <l xml:id="SS.2.1.28"> hanti sandhigataḥ sandhīn śūlaśophau karoti ca |
            asthiśoṣaṃ prabhedaṃ ca kuryāc chūlaṃ ca tacchritaḥ || </l>

          <l xml:id="SS.2.1.29"> tathā majjagate ruk ca na kadācit praśāmyati | apravṛttiḥ
            pravṛttirvā vikṛtā śukrage 'nile || </l>

          <l xml:id="SS.2.1.30"> hastapādaśirodhātūṃs tathā saṃcarati kramāt | vyāpnuyādvā
            'khilaṃ dehaṃ vāyuḥ sarvagato nṛṇām || </l>

          <l xml:id="SS.2.1.31"> stambhanākṣepaṇasvāpaśophaśūlāni sarvagaḥ | sthāneṣūkteṣu
            saṃmiśraḥ saṃmiśrāḥ kurute rujaḥ || </l>

          <l xml:id="SS.2.1.32"> kuryād avayavaprāpto mārutas tv amitān gadān |
            dāhasaṃtāpamūrcchāḥ syurvāyau pittasamanvite || </l>

          <l xml:id="SS.2.1.33"> śaity aśophagurutvāni tasminn eva kaphāvṛte | sūcībhir
            iva nistodaḥ sparśadveṣaḥ prasuptatā || </l>

          <l xml:id="SS.2.1.34"> śeṣāḥ pittavikārāḥ syur mārute śoṇitānvite | prāṇe
            pittāvṛte chardirdāhaś caivopajāyate || </l>

          <l xml:id="SS.2.1.35"> daurbalyaṃ sadanaṃ tandrā vaivarṇyaṃ ca kaphāvṛte | udāne
            pittasaṃyukte mūrcchādāhabhramaklamāḥ || </l>

          <l xml:id="SS.2.1.36"> asvedaharṣau mando 'gniḥ śītastambhau kaphāvṛte | samāne
            pittasaṃyukte svedadāhauṣṇyamūrcchanam || </l>

          <l xml:id="SS.2.1.37"> kaphādhikaṃ ca viṇmūtraṃ romaharṣaḥ kaphāvṛte | apāne
            pittasaṃyukte dāhauṣṇye syād asṛgdaraḥ || </l>

          <l xml:id="SS.2.1.38"> adhaḥkāyagurutvaṃ ca tasminn eva kaphāvṛte | vyāne
            pittāvṛte dāho gātravikṣepaṇaṃ klamaḥ || </l>

          <l xml:id="SS.2.1.39"> gurūṇi sarvagātrāṇi stambhanaṃ cāsthiparvaṇām | liṅgaṃ
            kaphāvṛte vyāne ceṣṭāstambhas tathaiva ca || </l>

          <l xml:id="SS.2.1.40"> prāyaśaḥ sukumārāṇāṃ mithyā++āhāravihāriṇām |
            rogādhvapramadāmadyavyāyāmaiś cātipīḍanāt || </l>

          <l xml:id="SS.2.1.41"> ṛtusātmyaviparyāsāt snehādīnāṃ ca vibhramāt | avyavāye
            tathā sthūle vātaraktaṃ prakupyati ||) </l>

          <l xml:id="SS.2.1.42"> hastyaśvoṣṭrairgacchato 'nyaiś ca vāyuḥ kopaṃ yātaḥ
            kāraṇaiḥ sevitaiḥ svaiḥ | tīkṣṇoṣṇāmlakṣāraśākādibhojyaiḥ saṃtāpādyair bhūyasā
            sevitaiś ca || </l>

          <l xml:id="SS.2.1.43"> kṣipraṃ raktaṃ duṣṭimāyāni tac ca vāyor mārgaṃ
            saṃruṇaddhyāśu yātaḥ | kruddho 'tyarthaṃ mārgarodhāt sa vāyur atyudriktaṃ
            dūṣayed raktam āśu || </l>

          <l xml:id="SS.2.1.44"> tat saṃpṛktaṃ vāyunā dūṣitena tat prāvalyād ucyate
            vātaraktam | tadvat pittaṃ dūṣitenāsṛjā++āktaṃ śleṣmā duṣṭo
            dūṣitenāsṛjā++āktaḥ || </l>

          <l xml:id="SS.2.1.45"> sparśodvignau todabhedapraśoṣasvāpopetau vātaraktena
            pādau | pittāsṛgbhyām ugradāhau bhavet āmatyarthoṣṇau raktaśophau mṛdū ca || </l>

          <l xml:id="SS.2.1.46"> kaṇḍūmantau śvetaśītau saśophau pīnastabdhau śleṣmaduṣṭe
            tu rakte | sarvairduṣṭe śoṇite cāpi doṣāḥ svaṃ svaṃ rūpaṃ pādayor darśayanti
            || </l>

          <l xml:id="SS.2.1.47"> prāgrūpe śithilau svinnau śītalau saviparyayau |
            vaivarṇyatodasuptatvagurutvauṣasamanvitau || </l>

          <l xml:id="SS.2.1.48"> pādayor mūlam āsthāya kadācid dhastayor api | āsvor viṣam
            iva kruddhaṃ tad deham anusarpati || </l>

          <l xml:id="SS.2.1.49"> ājānusphuṭitaṃ yac ca prabhinnaṃ prasrutaṃ ca yat |
            upadravaiś ca yajyuṣṭaṃ prāṇam āṃsakṣayādibhiḥ || </l>

          <l xml:id="SS.2.1.50"> śoṇitaṃ tad asādhyaṃ syād yāpyaṃ saṃvatsarotthitam | yadā
            tu dhamanīḥ sarvāḥ kupito 'bhyeti mārutaḥ || </l>

          <l xml:id="SS.2.1.51"> tad ākṣipaty āśu muhur muhur dehaṃ muhuś caraḥ |
            muhurmuhus tad ākṣepād ākṣepaka iti smṛtaḥ || </l>

          <l xml:id="SS.2.1.52"> so 'patānakasaṃjño yaḥ pātayaty antarā 'ntarā |
            kaphānvito bhṛśaṃ vāyus tāsv eva yadi tiṣṭhati || </l>

          <l xml:id="SS.2.1.53"> sa daṇḍavat stambhayati kṛcchro daṇḍāpatānakaḥ |
            hanugrahas tadā 'tyarthaṃ kṛcchrān niṣevate || </l>

          <l xml:id="SS.2.1.54"> dhanus tulyaṃ named yas tu sa dhanuḥstambhasaṃjñakaḥ |
            aṅgulīgulphajaṭharahradvakṣogalasaṃśritaḥ || </l>

          <l xml:id="SS.2.1.55"> snāyupratānam anilo yadā++ākṣipati vegavān |
            viṣṭabdhākṣaḥ stabdhahanur bhagnapārśvaḥ kaphaṃ vaman || </l>

          <l xml:id="SS.2.1.56"> abhyantaraṃ dhanur iva yadā namati mānavaḥ | tadā
            'syābhyantarāyāmaṃ kurute māruto balī || </l>

          <l xml:id="SS.2.1.57"> bāhyasnāyupratānastho bāhyāyāmaṃ karoti ca | tamasād hy
            aṃbudhāḥ prāhur vakṣaḥkaṭhyūrubhañjanam || </l>

          <l xml:id="SS.2.1.58"> kaphapittānvito vāyur vāyur eva ca kevalaḥ | kuryād
            ākṣepakaṃ tv anyaṃ caturtham abhighātajam || </l>

          <l xml:id="SS.2.1.59"> garbhapātanimittaś ca śoṇitātisravāc ca yaḥ |
            abhighātanimittaś ca na sidhyaty apatānakaḥ || </l>

          <l xml:id="SS.2.1.60"> adhogamāḥ satiryaggā dhamanīrūrdhvadehagāḥ | yadā
            prakupito 'tyarthaṃ mātariśvā prapadyate || </l>

          <l xml:id="SS.2.1.61"> tadā 'nyatarapakṣasya sandhibandhān vimokṣayan | hanti
            pakṣaṃ tamāhurhi pakṣāghātaṃ bhiṣagvarāḥ || </l>

          <l xml:id="SS.2.1.62"> yasya kṛtsnaṃ śarīrārdham akarmaṇyam acetanam | tataḥ
            patasyasūn vā 'pi tyajaty anilapīḍitaḥ || </l>

          <l xml:id="SS.2.1.63"> śuddhavātahataṃ pakṣaṃ kṛcchrasādhyatamaṃ viduḥ |
            sādhyamanyena saṃsṛṣṭamasādhyaṃ kṣayahetukam || </l>

          <l xml:id="SS.2.1.64"> vāyur ūrdhvaṃ vrajet sthānāt kupito hṛdayaṃ śiraḥ |
            śaṅkhau ca pīḍayaty aṅgāny ākṣipen namayec ca saḥ || </l>

          <l xml:id="SS.2.1.65"> nimīlitākṣo niśceṣṭaḥ stabdhākṣo vā 'pi kūjati |
            nirucchvāso 'thavā kṛcchrād ucchvasyān naṣṭacetanaḥ || </l>

          <l xml:id="SS.2.1.66"> svasthaḥ syād dhṛdaye mukte hyāvṛte tu pramuhyati |
            kaphānvitena vātena jñeya eṣo 'patantrakaḥ || </l>

          <l xml:id="SS.2.1.67"> divāsvapnāsanasthānavivṛtādhvanirīkṣaṇaiḥ | manyāstambhaṃ
            prakurute sa eva śleṣmaṇā++āvṛtaḥ || </l>

          <l xml:id="SS.2.1.68"> (garbhiṇīsūtikābālavṛddhakṣīṇeṣv asṛkkṣaye |) uccair
            vyāharato 'tyarthaṃ svādataḥ kaṭhināni vā | hasato jṛmbhato bhārād viṣam
            ācchayanād api || </l>

          <l xml:id="SS.2.1.69"> śironāsauṣṭhacibukalalāṭekṣaṇasandhigaḥ | ardayitvā 'nilo
            vaktramarditaṃ janayaty ataḥ || </l>

          <l xml:id="SS.2.1.70"> vakrībhavati vaktrārdhaṃ grīvā cāpy apavartate | śiraś
            calati vāksaṅgo netrādīnāṃ ca vaikṛtam || </l>

          <l xml:id="SS.2.1.71"> grīvācibukadantānāṃ tasmin pārśve tu vedanā | yasyāgrajo
            romaharṣo vepathurnetramāvilam || </l>

          <l xml:id="SS.2.1.72"> vāyur ūrdhvaṃ tvaci svāpas todo manyāhanugrahaḥ | tam
            arditam iti grāhurvyādhiṃ vyādhiviśāradāḥ || </l>

          <l xml:id="SS.2.1.73"> kṣīṇasyānimiṣākṣasya prasaktaṃ saktabhāṣiṇaḥ | na
            sidhyaty arditaṃ bāḍhaṃ trivarṣaṃ vepanasya ca || </l>

          <l xml:id="SS.2.1.74"> pārṣṇipratyaṅgulīnāṃ tu kaṇḍarā yā 'nilārditāḥ | sakthnaḥ
            kṣepaṃ nigṛhṇīyād gṛdhrasīti hi sā smṛtā || </l>

          <l xml:id="SS.2.1.75"> talapratyaṅgulīnāṃ tu kaṇḍarā bāhupṛṣṭhataḥ | bāhvoḥ
            karmakṣayakarī viśvācīti hi sā smṛtā || </l>

          <l xml:id="SS.2.1.76"> vātaśoṇitajaḥ śopho jānumadhye mahārujaḥ | śiraḥ
            kroṣṭukapūrvaṃ tu sthūlaḥ kroṣṭukamūrdhavat || </l>

          <l xml:id="SS.2.1.77"> vāyuḥ kaṭyāṃ sthitaḥ sakthnaḥ kaṇḍarāmākṣipedyadā |
            svañjas tadā bhavejjantuḥ, paṅguḥ sakthnordvayor vadhāt || </l>

          <l xml:id="SS.2.1.78"> prakrāman vepate yas tu svañjanniva ca gacchati |

          <l xml:id="SS.2.15.1"> athāto bhagnānāṃ nidānaṃ vyākhyāsyāmaḥ || </l>

          <l xml:id="SS.2.15.2"> yathovāca bhagavān dhanvantariḥ || </l>

          <l xml:id="SS.2.15.3"> patanapīḍanaprahārakṣepaṇavyālamṛgadaśanaprabhṛtibhir
            abhighātaviśeṣair anekavidhamasthnāṃ bhaṅgam upadiśanti || </l>

          <l xml:id="SS.2.15.4"> tatra bhaṅga(gna)jātamanekavidhamanusāryamāṇaṃ
            dvividhamevopapadyate sandhimukataṃ kāṇḍabhagnaṃ ca | tatra ṣaḍvidhaṃ
            sandhimuktaṃ dvādaśavidhaṃ kāṇḍabhagnaṃ bhavati || </l>

          <l xml:id="SS.2.15.5"> tatra sandhimuktam utpiṣṭaṃ viśliṣṭaṃ vivartitam
            avakṣiptam atikṣiptaṃ tiryakkṣiptam iti ṣaḍvidham || </l>

          <l xml:id="SS.2.15.6"> tatra prasāraṇākuñcanavivartanākṣepaṇāśaktirugrarujatvaṃ
            sparśāsahatvaṃ ceti sāmānyaṃ sandhimuktalakṣaṇamuktam || </l>

          <l xml:id="SS.2.15.7"> vaiśeṣikaṃ tūtpiṣṭe sandhāvumayataḥ śopho
            vedanāprādurbhāvo viśeṣataś ca nānāprakārā vedanā rātrau prādurbhavanti;
            viśliṣṭe 'lpaḥ śopho vedanāsātatyaṃ sandhivikriyā ca; vivartite tu
            sandhipārśvāpagamanādviṣamāṅgatā vedanā ca; avakṣipte
            sandhiviśleṣastīvrarujatvaṃ ca; atikṣipte dvayoḥ sandhyasthnoratikrāntatā
            vedanā ca; tiryakkṣipte tv ekāsthipārśvāpagamanamatyarthaṃ vedanā ceti || </l>

          <l xml:id="SS.2.15.8"> kāṇḍabhagnamata ūrdhvaṃ vakṣyāmaḥ karkaṭakam aśvakarṇaṃ
            cūrṇitaṃ piccitam asthicchallitaṃ kāṇḍabhagnaṃ majjānugatam atipātitaṃ vakraṃ
            chinnaṃ pāṭitaṃ sphuṭitam iti dvādaśavidham || </l>

          <l xml:id="SS.2.15.9"> śvayathubāhulyaṃ
            spandanavivartanasparśāsahiṣṇutvamavapīḍyamāne śabdaḥ srastāṅgatā
            vividhavedanāprādurbhāvaḥ sarvāsvavasthāsu na śarmalābha iti samāsena
            kāṇḍabhagnalakṣaṇamuktam || </l>

          <l xml:id="SS.2.15.10"> viśeṣas tu saṃmūḍhamubhayato 'sthi madhye bha(la)gnaṃ
            granthirivonnataṃ karkaṭakam aśvakarṇavadudgatamaśvakarṇakaṃ, spṛśyamānaṃ
            śabdavaccūrṇitamavagacchet, piccitaṃ pṛthutāṃ gatamanalpaśophaṃ, pārśvayor
            asthi hīnodgatamasthicchalitaṃ, vellate prakampamānaṃ kāṇḍabhagnam,
            asthyavayavo 'sthimadhyamanupraviśya majjānamunnahyatīti majjānugatam, asthi
            niḥśeṣataśchinnamatipātitam, ābhugnamavimuktāsthi vakram,
            anyatarapārśvāvaśiṣṭaṃ chinnaṃ, pāṭitamaṇubahuvidāritaṃ vedanāvacca,
            śūkapūrṇamivādhmātaṃ vipulaṃ visphuṭitaṃ sphuṭitam iti || </l>

          <l xml:id="SS.2.15.11"> teṣu cūrṇitacchinnātipātitamajjānugatāni kṛcchrasādhyāni
            kṛśavṛddhabālānāṃ kṣatakṣīṇakuṣṭhiśvāsināṃ sandhyupagataṃ ceti || </l>

          <l xml:id="SS.2.15.12"> bhavanti cātra bhinnaṃ kapālaṃ kaṭhyāṃ tu sandhimuktaṃ
            tathā cyutam | jaghanaṃ prati piṣṭaṃ ca varjayettaccikitsakaḥ || </l>

          <l xml:id="SS.2.15.13"> asaṃśliṣṭaṃ kapālaṃ tu lalāṭe cūrṇitaṃ ca yat | bhagnaṃ
            stanāntare śaṅkhe pṛṣṭhe mūrdhni ca varjayet || </l>

          <l xml:id="SS.2.15.14"> ādito yac ca durjātamasthi sandhirathāpi vā |
            samyagyamitamapyasthi durnyāsāddurnibandhanāt || </l>

          <l xml:id="SS.2.15.15"> saṅkhobhādvā 'pi yadgacchedvikriyāṃ tac ca varjayet |
            madhyasya vayaso 'vasthāstisro yāḥ parikīrtitāḥ || </l>

          <l xml:id="SS.2.15.16"> tatra sthiro bhavejjanturupakrānto vijānatā |
            taruṇāsthīni namyante bhajyante nalakāni tu || </l>

          <l xml:id="SS.2.15.17"> kapālān vibhidyante sphuṭanti rucakāni ca ||</l>

            'dhyāyaḥ ||15 ||</p>

          <head>ṣoḍaśo 'dhyāyaḥ |</head>

          <l xml:id="SS.2.16.1"> athāto mukharogāṇāṃ nidānaṃ vyākhyāsyāmaḥ || </l>

          <l xml:id="SS.2.16.2"> yathovāca bhagavān dhanvantariḥ || </l>

          <l xml:id="SS.2.16.3"> mukharogāḥ pañcaṣaṣṭirbhavanti saptasvāyataneṣu |
            tatrāyatanāni oṣṭhau, dantamūlāni, dantāḥ, jihvā, tālu, kaṇṭhaḥ, sarvāṇi ceti
            | tatrāṣṭāvoṣṭhayoḥ, pañcadaśa dantamūleṣu, aṣṭau danteṣu, pañca jihvāyāṃ,
            nava tāluni, saptadaśa kaṇṭhe, trayaḥ sarveṣvāyataneṣu || </l>

          <l xml:id="SS.2.16.4"> tatrauṣṭhaprakopā
            vātapittaśleṣmasannipātaraktamāṃsamedobhighātanimittāḥ || </l>

          <l xml:id="SS.2.16.5"> karkaśau paruṣau stabdhau kṛṣṇau tīvraruganvitau |
            dālyete paripāṭhyete hyoṣṭhau mārutakopataḥ || </l>

          <l xml:id="SS.2.16.6"> ācitau piḍākābhis tu sarṣapākṛtibhir bhṛśam |
            sadāhapākasaṃsrāvau nīlau pītau ca pittataḥ || </l>

          <l xml:id="SS.2.16.7"> savarṇābhis tu cīyete piḍakābhir avedanau | kaṇḍūmantau
            kaphācchūnau picchilau śītalau gurū || </l>

          <l xml:id="SS.2.16.8"> sakṛt kṛṣṇau sakṛt pītau sakṛcchvetau tathaiva ca |
            sannipātena vijñeyāvanekapiḍikācitau || </l>

          <l xml:id="SS.2.16.9"> kharjūraphalavarṇābhiḥ piḍakābhiḥ samācitau |
            raktopasṛṣṭau rudhiraṃ sravataḥ śoṇitaprabhau || </l>

          <l xml:id="SS.2.16.10"> māsaduṣṭau gurū sthūlau māṃsapiṇḍavadudgatau | jantavaś
            cātra mūrcchanti sṛkkasyobhayato mukhāt || </l>

          <l xml:id="SS.2.16.11"> medasā ghṛtamaṇḍābhau kaṇḍūmantau sthirau mṛdū || acchaṃ
            sphaṭikasaṅkāśamāsrāvaṃ sravayo gurū || </l>

          <l xml:id="SS.2.16.12"> kṣatajābhau vidīryete pāṭhyete cābhighātataḥ | grathitau
            ca samākhyātāvoṣṭhau kaṇḍūsamanvitau || </l>

          <l xml:id="SS.2.16.13"> dantamūlagatās tu śītādo, dantapuppuṭako, dantaveṣṭakaḥ,
            śauṣiro, mahāśauṣiraḥ, paridara, upakuśo, dantavaidarbho, vardhanaḥ,
            adhimāṃso, nāḍyaḥ pañceti || </l>

          <l xml:id="SS.2.16.14"> śoṇitaṃ dantaveṣṭebhyo yasyākasmāt pravartate |
            durgandhīni sakṛṣṇāni prakledīni mṛdūni ca || </l>

          <l xml:id="SS.2.16.15"> dantamāṃsāni śīryante pacanti ca parasparam | śītādo
            nāma sa vyādhiḥ kaphaśoṇitasaṃbhavaḥ || </l>

          <l xml:id="SS.2.16.16"> dantayostriṣu vā yasya śvayathuḥ sarujo mahān |
            dantapuppuṭako jñeyaḥ kapharaktanimittajaḥ || </l>

          <l xml:id="SS.2.16.17"> sravanti pūyarudhiraṃ calā dantā bhavanti ca |
            dantaveṣṭaḥ sa vijñeyo duṣṭaśoṇitasaṃbhavaḥ || </l>

          <l xml:id="SS.2.16.18"> śvayathurdantamūleṣu rujāvān kapharaktajaḥ | lālāsrāvī
            sa vijñeyaḥ kaṇḍūmāñ śauṣiro gadaḥ || </l>

          <l xml:id="SS.2.16.19"> dantāścalanti veṣṭebhyastālu cāpy avadīryate |
            dantamāṃsāni pacyante mukhaṃ ca paripīḍyate || </l>

          <l xml:id="SS.2.16.20"> yasmin sa sarvajo vyādhirmahāśauṣirasaṃjñakaḥ |
            dantamāṃsāni śīryante yasmin ṣṭhīvati cāpy asṛk || </l>

          <l xml:id="SS.2.16.21"> pittāsṛkkaphajo vyādhirjñeyaḥ paridaro hi saḥ | ceṣṭeṣu
            dāhaḥ pākaś ca tebhyo dantāś calanti ca || </l>

          <l xml:id="SS.2.16.22"> āghaṭṭitāḥ prasravanti śoṇitaṃ mandavedanāḥ | ādhmāyante
            srute rakte mukhaṃ pūti ca jāyate || </l>

          <l xml:id="SS.2.16.23"> yasminnupakuśaḥ sa syāt pittaraktakṛto gadaḥ | ghṛṣṭeṣu
            dantamūleṣu saṃrambho jāyate mahān || </l>

          <l xml:id="SS.2.16.24"> bhavanti ca calā dantāḥ sa vaidarbho 'bhighātajaḥ |
            mārutenādhiko danto jāyate tīvravedanaḥ || </l>

          <l xml:id="SS.2.16.25"> vardhanaḥ sa mato vyādhirjāte ruk ca praśāmyati |
            hānavye paścime dante mahāñ chotho mahārujaḥ || </l>

          <l xml:id="SS.2.16.26"> lālāsrāvī kaphakṛto vijñeyaḥ so 'dhimāṃsakaḥ |
            dantamūlagatā nāḍyaḥ pañca jñeyā yatheritāḥ || </l>

          <l xml:id="SS.2.16.27"> dantagatās tu dālanaḥ, krimidantako, dantaharṣo,
            bhañjanakaḥ, dantaśarkarā, kapālikā, śyāvadantako, hanumokṣaś ceti || </l>

          <l xml:id="SS.2.16.28"> dālyante bahudhā dantā yasmiṃstīvraruganvitāḥ | dālanaḥ
            sa iti jñeyaḥ sadāgatinimittajaḥ || </l>

          <l xml:id="SS.2.16.29"> kṛṣṇaścchidrī calaḥ srāvī sasṃrambho mahārujaḥ |
            animittarujo vātādvijñeyaḥ kṛmidantakaḥ || </l>

          <l xml:id="SS.2.16.30"> śītamuṣṇaṃ ca daśanāḥ sahante sparśanaṃ na ca | yasya
            taṃ dantaharṣaṃ tu vyādhiṃ vidyāt samīraṇāt || </l>

          <l xml:id="SS.2.16.31"> vaktraṃ vakraṃ bhaved yasmin dantabhaṅgaś ca tīvraruk |
            kaphavātakṛto vyādhiḥ sa bhañjanakasaṃjñitaḥ || </l>

          <l xml:id="SS.2.16.32"> śarkareva sthirībhūto malo danateṣu yasya vai | sā
            dantānāṃ guṇaharī vijñeyā dantaśarkarā || </l>

          <l xml:id="SS.2.16.33"> dalanti dantavalkāni yadā śarkarayā saha | jñeyā
            kapālikā saiva daśanānāṃ vināśinī || </l>

          <l xml:id="SS.2.16.34"> yo 'sṛṅniśreṇa pittena dagdho dantas tv aśeṣataḥ |
            śyāvatāṃ nīlatāṃ vā 'pi gataḥ sa śyāvadantakaḥ || </l>

          <l xml:id="SS.2.16.35"> vātena tais tair bhāvais tu hanusandhirvidaṃhataḥ |
            hanumokṣa iti jñeyo vyādhir arditalakṣaṇaḥ || </l>

          <l xml:id="SS.2.16.36"> jihvāgatāstu kaṇṭakāstrividhās tribhir doṣaiḥ, alāsa,
            upajihvikā ceti || </l>

          <l xml:id="SS.2.16.37"> jihvā 'nilena sphuṭitā prasuptā bhavec ca
            śākacchadanaprakāśā | pittena pītā paridahyate ca citā saraktair api kaṇṭakaiś
            ca | kaphena gurvī bahalā citā ca māṃsodgamaiḥ śālmalikaṇṭakābhaiḥ || </l>

          <l xml:id="SS.2.16.38"> jihvātale yaḥ śvayathuḥ pragāḍhaḥ so 'lāsasaṃjñaḥ
            kapharaktamūrtiḥ | jihvāṃ sa tu stambhayati pravṛddho mūle tu jihvā bhṛśam eti
            pākam || </l>

          <l xml:id="SS.2.16.39"> jihvāgrarūpaḥ śvayathur hi jihvām unnamya jātaḥ
            kapharaktayoniḥ | prasekakaṇḍūparidāhayuktā prakathyate 'sāv upajihviketi || </l>

          <l xml:id="SS.2.16.40"> tālugatās tu galaśuṇḍikā, tuṇḍikerī, adhruṣaḥ,
            kacchapaḥ, arbudaṃ, māṃsasaṅghātaḥ, tālupuppuṭaḥ, tāluśoṣaḥ, tālupāka iti || </l>

          <l xml:id="SS.2.16.41"> śleṣmāsṛgbhyāṃ tālumūlāt pravṛddho dīrghaḥ śopho
            dhmātabastiprakāśaḥ | tṛṣṇākāsaśvāsakṛt saṃpradiṣṭo vyādhir vaidyaiḥ
            kaṇṭhaśuṇḍīti nāmnā || </l>

          <l xml:id="SS.2.16.42"> śophaḥ stabdho lohiṃtastāludeśe raktājñeyaḥ so 'dhruṣo
            rugjvarāḍhyaḥ || </l>

          <l xml:id="SS.2.16.43"> kūrmotsanno 'vedano 'śīghrajanmā 'rakto jñeyaḥ kacchapaḥ
            śleṣmaṇā syāt | padmākāraṃ tālumadhye tu śophaṃ vidyād raktād arbudaṃ
            proktaliṅgam || </l>

          <l xml:id="SS.2.16.44"> duṣṭaṃ māṃsaṃ śleṣmaṇā nīrujaṃ ca tālvantaḥsthaṃ
            māṃsasaṅghātam āhuḥ | nīruk sthāyī kolamātraḥ kaphāt syān medoyuktāt
            puppuṭastāludeśe || </l>

          <l xml:id="SS.2.16.45"> śoso 'tyarthaṃ dīryate cāpi tāluḥ śvāso vātāttāluśoṣaḥ
            sapittāt | pittaṃ kuryāt pākam atyarthaghoraṃ tāluny enaṃ tālupākaṃ vadanti || </l>

          <l xml:id="SS.2.16.46"> kaṇṭhagatās tu rohiṇyaḥ pañca, kaṇṭhaśālūkam, adhijihvo,
            balayo, balāsa, ekavṛndo, vṛndaḥ, śataghnī, gilāyuḥ, galavidradhiḥ, galaughaḥ,
            svaraghno, māṃsatāno, vidārī ceti || </l>

          <l xml:id="SS.2.16.47"> gale 'nilaḥ pittakaphau ca mūrcchitau pṛthak samastāś ca
            tathaiva śoṇitam | pradūṣya māṃsaṃ galarodhino 'ṅkurān sṛjanti yān sā 'suharā
            hi rohiṇī || </l>

          <l xml:id="SS.2.16.48"> jihvāṃ samantād bhṛśavedanā ye māṃsāṅkurāḥ
            kaṇṭhanirodhinaḥ syuḥ | tāṃ rohiṇīṃ vātakṛtāṃ vadanti vātātm
            akopadravagāḍhayuktām || </l>

          <l xml:id="SS.2.16.49"> kṣiprodgamā kṣipravidāhapākā tīvrajvarā pittanimittajā
            syāt | srotonirodhiny api mandapākā gurvī sthirā sā kaphasaṃbhavā vai || </l>

          <l xml:id="SS.2.16.50"> gambhīrapākā 'prativāravīryā tridoṣaliṅgā trayasaṃbhavā
            syāt | sphoṭācitā pittasamānaliṅgā 'sādhyā pradiṣṭā rudhirātmikeyam || </l>

          <l xml:id="SS.2.16.51"> kolāsthimātraḥ kaphasṃbhavo yo granthir gale
            kaṇṭakaśūkabhūtaḥ | kharaḥ sthiraḥ śastranipātasādhyas taṃ kaṇṭhaśālūkam iti
            bruvanti || </l>

          <l xml:id="SS.2.16.52"> jihvāgrarūpaḥ śvayathuḥ kaphāttu jihvāprabandhopari
            raktamiśrāt | jñeyo 'dhijihvaḥ khalu roga eṣa vivarjayed āgatapākam enam || </l>

          <l xml:id="SS.2.16.53"> balāsa evāyatamunnataṃ ca śophaṃ karotyannagatiṃ nivārya
            | taṃ sarvathaivāprativāravīryaṃ vivarjanīyaṃ valayaṃ vadanti || </l>

          <l xml:id="SS.2.16.54"> gale tu śophaṃ kurutaḥ pravṛddhau śleṣmānilau
            śvāsarujopapannam | marmacchidaṃ dustaram etad āhur balāsasṃjñaṃ nipuṇā
            vikāram || </l>

          <l xml:id="SS.2.16.55"> vṛttonnato yaḥ śvayathuḥ sadāhaḥ kaṇḍvanvito
            'pākyamṛdurguruś ca | nāmnaikavṛndaḥ parikīrtito 'sau vyādhir
            balāsakṣatajaprasūtaḥ || </l>

          <l xml:id="SS.2.16.56"> samunnataṃ vṛttam amandadāhaṃ tīvrajvaraṃ
            vṛndamudāharanti | taṃ cāpi pittakṣatajaprakopādvidyāt satodaṃ pavanāsrajaṃ tu
            || </l>

          <l xml:id="SS.2.16.57"> vartir ghanā kaṇṭhanirodhinī yā citā 'timātraṃ
            piśitaprarohaiḥ | nānārujocchrāyakarī tridoṣājjñeyā śataghnīva śataghny
            asādhyā || </l>

          <l xml:id="SS.2.16.58"> granthirgale tvāmalakāsthimātraḥ sthiro 'lparuk syāt
            kapharaktamūrtiḥ | saṃlakṣyate saktam ivāśanaṃ ca sa śastrasādhyas tu
            gilāyusaṃjñaḥ || </l>

          <l xml:id="SS.2.16.59"> sarvaṃ galaṃ vyāpya samutthito yaḥ śopho rujo yatra ca
            santi sarvāḥ | sa sarvadoṣo galavidradhis tu tasyaiva tulyaḥ khalu sarvajasya
            || </l>

          <l xml:id="SS.2.16.60"> śopho mahānannajalāvarodhī tīvrajvaro vātagaternihantā |
            kaphena jāto rudhirānvitena gale galaughaḥ parikīrtyate 'sau || </l>

          <l xml:id="SS.2.16.61"> yo 'tipratāmyan śvasiti prasaktaṃ bhinnasvaraḥ
            śuṣkavimuktakaṇṭhaḥ | kaphopadigdheṣv anilāyaneṣu jñeyaḥ sa rogaḥ śvasanāt
            svaraghnaḥ || </l>

          <l xml:id="SS.2.16.62"> pratānavān yaḥ śvayathuḥ sukaṣṭo galoparodhaṃ kurute
            krameṇa | sa māṃsatānaḥ kathito 'valambī prāṇapraṇut sarvakṛto vikāraḥ || </l>

          <l xml:id="SS.2.16.63"> sadāhatodaṃ śvayathuṃ saraktam antargale
            pūtiviśīrṇamāṃsam | pittena vidyād vadane vidārīṃ pārśve viśeṣāt sa tu yena
            śete || </l>

          <l xml:id="SS.2.16.64"> sarvasarās tu vātapittakaphaśoṇitanimittāḥ || </l>

          <l xml:id="SS.2.16.65"> sphoṭaiḥ satodair vadanaṃ samantād yasyācitaṃ sarvasaraḥ
            sa vātāt |
            <caesura/>
             raktaiḥ sadāhais tanubhiḥ sapītair yasyācitaṃ cāpi sa pittakopāt || </l>

          <l xml:id="SS.2.16.66"> kaṇḍūyutair alparujaiḥ sarvaṇair yasyācitaṃ cāpi sa vai
            kaphena |
            <caesura/>
             raktena pittodita eka eva kaiś cit pradiṣṭo mukhapākasaṃjñaḥ ||</l>
