          <l xml:id="SS.3.1.3">
            <label>1938 ed. SS.3.1.3</label>
            <caesura/>
             sarvabhūtānāṃ kāraṇam akāraṇaṃ sattvarajastamolakṣaṇam aṣṭarūpam akhilasya
            jagataḥ sambhavahetur avyaktaṃ nāma | tad ekaṃ bahūnāṃ kṣetrajñānām
            adhiṣṭhānaṃ samudra ivaudakānāṃ bhāvānām ||3|| </l>

          <l xml:id="SS.3.1.4">
            <label>1938 ed. SS.3.1.4</label>
            <caesura/>
             tasmād avyaktān mahān utpadyate talliṅga eva | talliṅgāc ca mahatas
            tallakṣaṇa evāhaṅkāra utpadyate; sa trividho vaikārikas taijaso bhūtādir iti |
            tatra vaikārikād ahaṅkārāt taijasasahāyāt tallakṣaṇāny evaikādaśendriyāṇy
            utpadyante,
            <caesura/>
            tad yathā --
            <caesura/>
             śrotratvakcakṣurjihvāghrāṇavāgghastopasthapāyupādamanāṃsīti; tatra pūrvāṇi
            pañca buddhīndriyāṇi, itarāṇi pañca karmendriyāṇi, ubhayātmakaṃ manaḥ;
            bhūtāder api taijasasahāyāt tallakṣaṇāny eva pañcatanmātrāṇy utpadyante --
            śabdatanmātraṃ, sparśatanmātraṃ, rūpatanmātraṃ, rasatanmātraṃ, gandhatanmātram
            iti; teṣāṃ viśeṣāḥ- śabdasparśarūparasagandhāḥ; tebhyo bhūtāni-
            vyomānilānalajalorvyaḥ; evam eṣā tattvacaturviṃśatir vyākhyātā ||4|| </l>

          <l xml:id="SS.3.1.5">
            <label>1938 ed. SS.3.1.5</label>
            <caesura/>
             tatra buddhīndriyāṇāṃ śabdādayo viṣayāḥ; karmendriyāṇāṃ yathāsaṅkhyaṃ
            vacanādānānandavisargaviharaṇāni ||5|| </l>

          <l xml:id="SS.3.1.6">
            <label>1938 ed. SS.3.1.6</label>
            <caesura/>
             avyaktaṃ mahānahaṅkāraḥ pañcatanmātrāṇi cetyaṣṭau prakṛtayaḥ; śeṣāḥ ṣoḍaśa
            vikārāḥ ||6|| </l>

          <l xml:id="SS.3.1.7">
            <label>1938 ed. SS.3.1.7</label>
            <caesura/>
             svaḥ svaś caiṣāṃ viṣayo 'dhibhūtaṃ; svayam adhyātmam; adhidaivatam -- atha
            buddher brahmā, ahaṅkārasyeśvaraḥ, manasaś candramāḥ, diśaḥ śrotrasya, tvaco
            vāyuḥ, sūryaś cakṣuṣaḥ, rasanasyāpaḥ, pṛthivī ghrāṇasya, vāco 'gniḥ, hastayor
            indraḥ, pādayor viṣṇuḥ, pāyor mitraḥ, prajāpatir upasthasyeti ||7|| </l>

          <l xml:id="SS.3.1.8">
            <label>1938 ed. SS.3.1.8</label>
            <caesura/>
             tatra sarva evācetana eṣa vargaḥ, puruṣaḥ pañcaviṃśatitamaḥ
            kāryakāraṇasaṃyuktaś cetayitā bhavati | saty apy acaitanye pradhānasya
            puruṣakaivalyārthaṃ pravṛttim upadiśanti, kṣīrādīṃś cātra hetūn udāharanti
            ||8|| </l>

          <l xml:id="SS.3.1.9">
            <label>1938 ed. SS.3.1.9</label>
            <caesura/>
             ata ūrdhvaṃ prakṛtipuruṣayoḥ sādharmyavaidharmye vyākhyāsyāmaḥ | tad yathā --
            ubhāv apy anādī, ubhāv apy anantau, ubhāv apy aliṅgau, ubhāv api nityau,
            ubhāv apy anaparau, ubhau ca sarvagatāv iti; ekā tu prakṛtir acetanā triguṇā
            bījadharmiṇī prasavadharmiṇyamadhyasthadharmiṇī ceti, bahavas tu
            puruṣāś cetanāvanto 'guṇā abījadharmāṇo 'prasavadharmāṇo madhyasthadharmāṇaś
            ceti ||9|| </l>

          <l xml:id="SS.3.1.10">
            <label>1938 ed. SS.3.1.10</label>
            <caesura/>
             tatra kāraṇānurūpaṃ kāryam iti kṛtvā sarva evaite viśeṣāḥ sattvarajastamomayā
            bhavanti; tadañjanatvāt tanmayatvāc ca tadguṇā eva puruṣā bhavantīty eke
            bhāṣante ||10|| </l>

          <l xml:id="SS.3.1.11">
            <label>1938 ed. SS.3.1.11</label>
            <caesura/>
             vaidyake tu --
            <caesura/>
             svabhāvamīśvaraṃ kālaṃ yadṛcchāṃ niyatiṃ tathā |
            <caesura/>
             pariṇāmaṃ ca manyante prakṛtiṃ pṛthudarśinaḥ ||11|| </l>

          <l xml:id="SS.3.1.12">
            <label>1938 ed. SS.3.1.12</label>
            <caesura/>
             tanmayāny eva bhūtāni tadguṇāny eva cādiśet |
            <caesura/>
             taiś ca tallakṣaṇaḥ kṛtsno bhūtagrāmo vyajanyata ||12|| </l>

          <l xml:id="SS.3.1.13">
            <label>1938 ed. SS.3.1.13</label>
            <caesura/>
             tasyopayogo 'bhihitaś cikitsāṃ prati sarvadā |
            <caesura/>
             bhūtebhyo hi paraṃ yasmān nāsti cintā cikitsite ||13||</l>

          <l xml:id="SS.3.1.14">
            <label>1938 ed. SS.3.1.14</label>
            <caesura/>
             yato 'bhihitaṃ -- “tatsambhavadravyasamūho bhūtādiruktaḥ” (sū.1); bhautikāni
            cendriyāṇyāyurvede varṇyante, tathendriyārthāḥ ||14|| </l>

          <l xml:id="SS.3.1.15">
            <label>1938 ed. SS.3.1.15</label>

            <caesura/>

            <caesura/>
             niyataṃ tulyayonitvān nānyenānyam iti sthitiḥ ||15|| </l>

          <l xml:id="SS.3.1.16">
            <label>1938 ed. SS.3.1.16</label>
            <caesura/>
             na cāyurvedaśāstreṣūpadiśyante sarvagatāḥ kṣetrajñā nityāś ca; asarvagateṣu
            ca kṣetrajñeṣu nity apuruṣakhyāpakān hetūnudāharanti; āyurvedaśāstreṣv
            asarvagatāḥ kṣetrajñā nityāś ca, tiryagyonimānuṣadeveṣu sañcaranti
            dharmādharmanimittaṃ; ta ete 'numānagrāhyāḥ paramasūkṣmāścetanāvantaḥ śāśvatā
            lohitaretasoḥ sannipāteṣv abhivyajyante, yato 'bhihitaṃ-
            ‘pañcamahābhūtaśarīrisamavāyaḥ puruṣaḥ’ (sū.1) iti; sa eṣa
            karmapuruṣaścikitsādhikṛtaḥ ||16|| </l>

          <l xml:id="SS.3.1.17">
            <label>1938 ed. SS.3.1.17</label>
            <caesura/>
             tasya sukhaduḥkhe icchādveṣau prayatnaḥ prāṇāpānāvunmeṣanimeṣau buddhirmanaḥ
            saṅkalpo vicāraṇā smṛtirvijñānamadhyavasāyo viṣayopalabdhiś ca guṇāḥ ||17|| </l>

          <l xml:id="SS.3.1.18">
            <label>1938 ed. SS.3.1.18</label>
            <caesura/>
             sāttvikāstu -- ānṛśaṃsyaṃ saṃvibhāgarucitā titikṣā satyaṃ dharma āstikyaṃ
            jñānaṃ buddhirmedhā smṛtirdhṛtiranabhiṣaṅgaś ca; rājasāstu --
            duḥkhabahulatā'ṭanaśīlatā'dhṛtirahaṅkāra ānṛtikatvam akāruṇyaṃ dambho māno
            harṣaḥ kāmaḥ krodhaś ca; tāmasāstu -- viṣāditvaṃ nāstikyamadharmaśīlatā
            buddhernirodho 'jñānaṃ durmedhastvam akarmaśīlatā nidrālutvaṃ ceti ||18|| </l>

          <l xml:id="SS.3.1.19">
            <label>1938 ed. SS.3.1.19</label>
            <caesura/>
             āntarikṣāḥ -- śabdaḥ śabdendriyaṃ sarvacchidrasamūho viviktatā ca; vāyavyāstu
            -- sparśaḥ sparśendriyaṃ sarvaceṣṭāsamūhaḥ sarvaśarīraspandanaṃ laghutā ca;
            taijasāstu -- rūpaṃ rūpendriyaṃ varṇaḥ santāpo bhrājiṣṇutā
            paktiramarṣastaikṣṇyaṃ śauryaṃ ca; āpyāstu -- raso rasanendriyaṃ
            sarvadravasamūho gurutā śaity aṃ sneho retaś ca; pārthivāstu -- gandho
            gandhendriyaṃ sarvamūrtasamūho gurutā ceti ||19|| </l>

          <l xml:id="SS.3.1.20">
            <label>1938 ed. SS.3.1.20</label>

             tatra sattvabahulamākāśaṃ, rajobahulo vāyuḥ, sattvarajobahulo 'gniḥ,
            sattvatamobahulā āpaḥ, tamobahulā pṛthivīti ||20|| </l>

          <l xml:id="SS.3.1.21ab">
