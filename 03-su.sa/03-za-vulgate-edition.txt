        <head>śārīrasthānam</head>

        <div n="1" type="adhyāya">
          <head>prathamo 'dhyāyaḥ |</head>
          <l xml:id="SS.3.1.1"> athātaḥ sarvabhūtacintāśārīraṃ vyākhyāsyāmaḥ || </l>

          <l xml:id="SS.3.1.2"> yathovāca bhagavān dhanvantariḥ || </l>

          <l xml:id="SS.3.1.3"> sarvabhūtānāṃ kāraṇam akāraṇaṃ sattvarajastamolakṣaṇam
            aṣṭarūpam akhilasya jagataḥ saṃbhavahetur avyaktaṃ nāma | tad ekaṃ bahūnāṃ
            kṣetrajñānām adhiṣṭhānaṃ samudra ivaudakānāṃ bhāvānām || </l>

          <l xml:id="SS.3.1.4"> tasmād avyaktān mahān utpadyate talliṅga eva; talliṅgāc ca
            mahatas tallakṣaṇa evāhaṅkāra utpadyate, sa trividho vaikārikastaijaso
            bhūtādir iti; tatra vaikārikādahaṅkārāttaijasasahāyāt tallakṣaṇāny
            enaikādaśendriyāṇy utpadyante, tad yathā ---
            śrotratvakcakṣurjihvāghrāṇavāgghastopasthapāyupādamanāṃsīti, tatra pūrvāṇi
            pañca buddhīndriyāṇi, itarāṇi pañca karmendriyāṇi, ubhayātmakaṃ manaḥ;
            bhūtāder api taijasasahāyāt tallakṣaṇāny eva pañcatanmātrāṇy utpadyante, tad
            yathā --- śabdatanmātraṃ, sparśatanmātraṃ, rūpatanmātraṃ, rasatanmātraṃ,
            gandhatanmātram iti; teṣāṃ viśeṣāṃ viśeṣāḥ śabdasparśarūparasagandhāḥ; tebhyo
            bhūtāni vyomānilānalajalorvyaḥ; evam eṣā tattvacaturviṃśatir vyākhyātā || </l>

          <l xml:id="SS.3.1.5"> tatra, buddhīndriyāṇāṃ śabdādayo viṣayāḥ; karmendriyāṇāṃ
            yathāsaṅkhyaṃ vacanādānānandavisargaviharaṇāni || </l>

          <l xml:id="SS.3.1.6"> avyaktaṃ mahānahaṅkāraḥ pañcatanmātrāṇi cety aṣṭau
            prakṛtayaḥ; śeṣāḥ ṣoḍaśa vikārāḥ || </l>

          <l xml:id="SS.3.1.7"> svaḥ svaś caiṣāṃ viṣayo 'dhibhūtaṃ; svayam adhyātmaṃ;
            adhidaivataṃ tu---budher brahyā, ahaṅkārasyeśvaraḥ, manasaś candramāḥ, diśaḥ
            śrotrasya, tvaco vāyuḥ, sūryaś cakṣuṣaḥ, rasanasyāpaḥ, pṛthivī ghrāṇasya, vāco
            'gniḥ, hastayor indraḥ, pādayor viṣṇuḥ, pāyor mitraḥ, prajāpatir upasthasyeti
            || </l>

          <l xml:id="SS.3.1.8"> tatra sarva evācetana eṣa vargaḥ, puruṣaḥ
            pañcaviṃśatitamaḥ kāryakāraṇasaṃyuktaś cetayitā bhavati | saty apy acaitanye
            pradhānasya puruṣakaivalyārthaṃ pravṛttim upadiśanti kṣīrādīṃś cātra hetūn
            udāharanti || </l>

          <l xml:id="SS.3.1.9"> ata ūrdhvaṃ prakṛtipuruṣayoḥ sādharmyavaidharmye
            vyākhyāsyāmaḥ | tadyathā --- ubhāv apy anādī, ubhāv apy anantau, ubhāv apy
            aliṅgau, ubhāv api nityau, ubhāv apy anaparau, ubhau ca sarvagatāv iti; ekā tu
            prakṛtir acetanā triguṇā bījadharmiṇī prasavadharmiṇy amadhyasthadharmiṇī
            ceti, bahavas tu puruṣāś cetanāvanto 'guṇā abījadharmāṇo 'prasavadharmāṇo
            madhyasthadharmāṇaś ceti || </l>

          <l xml:id="SS.3.1.10"> tatra kāraṇānurūpaṃ kāryam iti kṛtvā sarva evaite viśeṣāḥ
            sattvarajastamomayā bhavanti; tadañjanatvāt tanmayatvāc ca tadguṇā eva puruṣā
            bhavantīty eke bhāṣante || </l>

          <l xml:id="SS.3.1.11"> vaidyake tu ---
            <caesura/>
             svabhāvam īśvaraṃ kālaṃ yaddacchāṃ niyatiṃ tathā ||
            <caesura/>
             pariṇāmaṃ ca manyante prakṛtiṃ pṛthudarśinaḥ || </l>

          <l xml:id="SS.3.1.12"> tanmayāny eva bhūtāni tadguṇāny eva cādiśet ||
            <caesura/>
             taiś ca tallakṣaṇaḥ kṛtsno bhūtagrāmo vyajanyata || </l>

          <l xml:id="SS.3.1.13"> tasyopayogo 'bhihitaś cikitsāṃ prati sarvadā ||
            <caesura/>
             bhūtebhyo hi paraṃ yasmān nāsti cintā cikitsite || </l>

          <l xml:id="SS.3.1.14"> yato 'bhihitaṃ --- tat sambhavadravyasamūho bhūtādir
            uktaḥ; bhautikāni cendriyāṇy āyurvede varṇyante, tathendriyārthaḥ || </l>

          <l xml:id="SS.3.1.15"> bhavati cātra--- indriyeṇendiyārthaṃ tu svaṃ svaṃ gṛhṇāti
            mānavaḥ || niyataṃ tulyayonitvānnānyenānyam iti sthitiḥ || </l>

          <l xml:id="SS.3.1.16"> na cāyurvedaśāstreṣūpadiśyante sarvagatāḥ kṣetrajñā
            nityāś ca; asarvagateṣu ca kṣetrajñeṣu nityapuruṣakhyāpakān hetūn udāharanti;
            āyurvedaśāstrasiddhānteṣv asarvagatāḥ kṣetrajñā nityāś ca,
            tiryagyonimānuṣadeveṣu saṃcaranti dharmādharmanimittaṃ; ta ete 'numānagrāhyāḥ
            paramasūkṣmāścetanāvantaḥ śāśvatā lohitaretasoḥ sannipāteṣv abhivyajyante,
            yato 'bhihitaṃ --- pañcamahābhūtaśarīrisamavāyaḥ puruṣa iti; sa eṣa(eva?)
            karmapuruṣaś cikitsādhikṛtaḥ || </l>

          <l xml:id="SS.3.1.17"> tasya sukhaduḥkhe icchādveṣau prayatnaḥ prāṇāpānāv
            unmeṣanimeṣau buddhir manaḥ saṅkalpo vicāraṇā smṛtir vijñānam adhyavasāyo
            viṣayopalabdhiś ca guṇāḥ || </l>

          <l xml:id="SS.3.1.18"> sāttvikās tu --- ānṛśaṃsyaṃ saṃvibhāgarucitā titikṣā
            satyaṃ dharma āstikyaṃ jñānaṃ buddhir medhā smṛtir dhṛtir anabhiṣaṅgaś ca;
            rājasās tu --- duḥkhabahulatā'ṭanaśīlatā 'dhṛtirahaṅkāra ānṛtikatvam akāruṇyaṃ
            dambho māno harṣaḥ krodhaś ca; tāmasās tu --- viṣāditvaṃ nāstikyam
            adharmaśīlatā buddher nidho 'jñānaṃ durmedhastvam akarmaśīlatā nidrālutvaṃ
            ceti || </l>

          <l xml:id="SS.3.1.19"> āntarikṣāstu---śabdaḥ śabdendriyaṃ sarvacchidrasamūho
            viviktatā ca; vāyavyās tu --- sparśaḥ sparśondriyaṃ sarvaceṣṭāsamūhaḥ
            sarvaśarīraspandanaṃ laghutā ca; taijasās tu --- rūpaṃ rūpendriyaṃ varṇaḥ
            santāpo bhrājiṣṇutā paktir amarṣas taikṣṇyaṃ śauryaṃ ca; āpyās tu --- raso
            rasanendriyaṃ sarvadravasamūho gurutā śaityaṃ sneho retaś ca; pārthivās tu ---
            gandho gandhendriyaṃ sarvamūrtasamūho gurutā ceti || </l>

          <l xml:id="SS.3.1.20"> tatra sattvabahulam ākāśaṃ, rajobahulo vāyuḥ,
            sattvarajobahulo 'gniḥ, sattvatamobahulā āpaḥ, tamobahulā pṛthivīti || </l>

          <l xml:id="SS.3.1.21"> ślokau cātra bhavataḥ |</l>

          <lg>
            <l xml:id="SS.3.1.21ab">anyonyānupraviṣṭāni sarvāṇy etāni nirdiśet |</l>

            <l xml:id="SS.3.1.21cd">sve sve dravye tu sarveṣāṃ vyaktaṃ lakṣaṇamiṣyate
              ||</l>


          </lg>
          <l xml:id="SS.3.1.22"> aṣṭau prakṛtayaḥ proktā vikārāḥ ṣoḍaśaiva tu ||
            kṣetrajñaś ca samāsena svatantraparatantrayoḥ||</l>

          <p ana="trailer">iti suśrutasaṃhitāyāṃ śārīrasthāne sarvabhūtacintāśārīraṃ nāma
            prathamo 'dhyāyaḥ ||1 ||</p>

          <l xml:id="SS.3.2.4"> teṣu vātavarṇavedanaṃ vātena, pittavarṇavedanaṃ pittena,
